part 1

n = 1:100;
a_1 = 0.5;
b1 = [4.5];
a1 = [1 -a_1];
x1 = 3 * sin(2*pi*0.2*n);
y1 = filter(b1, a1, x1);
figure(1);
stem(n,y1);
title("responce when a= "+a_1);
xlabel('n');
ylabel('Amplitude');

part 3

n2 = 1:50;
step = ones(1,50);
% stem (n,step)
A1 = 16;
A2 = 4;
A3 = 5;
A4 = 3;
x_s1 = A1 * step;
x_s2 = A2 * step;
x_s3 = A3 * step;
x_s4 = A4 * step;
y_s1 = zeros(1, 50);
y_s1(1) = 0.5;
y_s2 = y_s1;
y_s3 = y_s1;
y_s4 = y_s1;
for (i = 2:50)
    y_s1(i) = 0.5*( y_s1(i-1) + x_s1(i)/y_s1(i-1));
end
for (i = 2:50)
    y_s2(i) = 0.5*( y_s2(i-1) + x_s2(i)/y_s2(i-1));
end
for (i = 2:50)
    y_s3(i) = 0.5*( y_s3(i-1) + x_s3(i)/y_s3(i-1));
end
for (i = 2:50)
    y_s4(i) = 0.5*( y_s4(i-1) + x_s4(i)/y_s4(i-1));
end
figure(5);
stem(n2,y_s1);
title("Square root responce for A= "+A1);
xlabel('n');
ylabel('Amplitude');

stem(n2,y_s2);
